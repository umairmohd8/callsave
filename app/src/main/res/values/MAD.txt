package com.example.counterapplication;

import androidx.appcompat.app.AppCompatActivity;

import android.os.Bundle;
import android.os.Handler;
import android.view.View;
import android.widget.Button;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity implements View.OnClickListener{
    TextView txtView;
    Button start, stop;
    Handler handler;
    Runnable runnable;
    int counter;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        start = findViewById(R.id.button2);
        stop = findViewById(R.id.button3);
        start.setOnClickListener(this);
        stop.setOnClickListener(this);
        counter = 0;
        handler = new Handler();
        txtView = findViewById(R.id.textView);
    }

    @Override
    public void onClick(View view) {
        if(view.equals(start)){
            handleCounter();
        }
        else
            handler.removeCallbacks(runnable);
    }

    private void handleCounter(){
        runnable = new Runnable() {
            @Override
            public void run() {
                counter++;
                txtView.setText(String.valueOf(counter));
                handler.postDelayed(this, 100);
            }
        };
        handler.post(runnable);
    }


}

package com.example.texttospeech;

import androidx.appcompat.app.AppCompatActivity;

import android.os.Build;
import android.os.Bundle;
import android.speech.tts.TextToSpeech;
import android.text.Editable;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;

import java.util.Locale;

public class MainActivity extends AppCompatActivity implements View.OnClickListener{

    private EditText txtView;
    private TextToSpeech textToSpeech;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        txtView = findViewById(R.id.textVIew);
        Button btn = findViewById(R.id.button);

        textToSpeech = new TextToSpeech(this, new TextToSpeech.OnInitListener() {
            @Override
            public void onInit(int i) {
                if(i != TextToSpeech.ERROR)
                    textToSpeech.setLanguage(Locale.US);
            }
        });
        btn.setOnClickListener(this);
    }

    @Override
    public void onClick(View view) {
        String text = txtView.getText().toString();
        if (text.isEmpty()) {
            text = "Please, Write something in the EditText...";
        }
//        textToSpeech.speak(text, TextToSpeech.QUEUE_FLUSH, null);
        textToSpeech.speak(text, TextToSpeech.QUEUE_FLUSH, null, null);
    }
}

package com.example.callandsave;

import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity implements View.OnClickListener{

        btn0 = findViewById(R.id.button12);
        btn0.setOnClickListener(this);
        btn1 = findViewById(R.id.button11);
        btn1.setOnClickListener(this);
        btn2 = findViewById(R.id.button10);
        btn2.setOnClickListener(this);
        btn3 = findViewById(R.id.button9);
        btn3.setOnClickListener(this);
        btn4 = findViewById(R.id.button8);
        btn4.setOnClickListener(this);
        btn5 = findViewById(R.id.button7);
        btn5.setOnClickListener(this);
        btn6 = findViewById(R.id.button6);
        btn6.setOnClickListener(this);
        btn7 = findViewById(R.id.button);
        btn7.setOnClickListener(this);
        btn8 = findViewById(R.id.button5);
        btn8.setOnClickListener(this);
        btn9 = findViewById(R.id.button4);
        btn9.setOnClickListener(this);
        btnDel = findViewById(R.id.button15);
        btnDel.setOnClickListener(this);
        btnSave = findViewById(R.id.button14);
        btnSave.setOnClickListener(this);
        btnCall = findViewById(R.id.button13);
        btnCall.setOnClickListener(this);
    }

    @Override
    public void onClick(View view) {
        if(view.equals(btn0))
            txtView.append("0");
        else if(view.equals(btn1))
            txtView.append("1");
        else if(view.equals(btn2))
            txtView.append("2");
        else if(view.equals(btn3))
            txtView.append("3");
        else if(view.equals(btn4))
            txtView.append("4");
        else if(view.equals(btn5))
            txtView.append("5");
        else if(view.equals(btn6))
            txtView.append("6");
        else if(view.equals(btn7))
            txtView.append("7");
        else if(view.equals(btn8))
            txtView.append("8");
        else if(view.equals(btn9))
            txtView.append("9");
        else if(view.equals(btnDel)) {
            String data = txtView.getText().toString();
            if(data.length()>0)
                txtView.setText(data.substring(0, data.length()-1));
            else
                txtView.setText("");
        }
        else if(view.equals(btnSave)){
            Intent it = new Intent(ContactsContract.Intents.Insert.ACTION);
            it.setType(ContactsContract.RawContacts.CONTENT_TYPE);
            it.putExtra(ContactsContract.Intents.Insert.PHONE, txtView.getText().toString());
            startActivity(it);
        }
        else if(view.equals(btnCall)){
            Intent it = new Intent(Intent.ACTION_DIAL);
            Uri uri = Uri.parse("tel:"+txtView.getText().toString());
            it.setData(uri);
            startActivity(it);
        }
    }
}
